-- CA (Captación)
--
-- Sustitucion de la select query
--
select selectquery  from queries
where id_layer=(select id_layer from layers where name='CA')

select replace(selectquery,' "eiel_c_abast_ca"."GEOMETRY"',' transform("eiel_c_abast_ca"."GEOMETRY", ?T) as "GEOMETRY"') from queries
where id_layer=(select id_layer from layers where name='CA')

update queries set selectquery=(select replace(selectquery,' "eiel_c_abast_ca"."GEOMETRY"',' transform("eiel_c_abast_ca"."GEOMETRY", ?T) as "GEOMETRY"') from queries
where id_layer=(select id_layer from layers where name='CA')) 
where id_layer=(select id_layer from layers where name='CA');
